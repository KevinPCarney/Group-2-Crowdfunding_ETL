campaign
-
cf_id int PK
contact_id int FK >- contacts.contact_id
company_name varchar(50)
blurb varchar(250)
goal float
pledged float
outcome varchar(20)
backers_count int
country varchar(10)
currency float
launched_at date
deadline date
category_id varchar(10) FK >- category.category_id
subcategory_id varchar(10) FK >- subcategory.subcategory_id
last_updated timestamp default=localtimestamp

contacts
-
contact_id int PK
first_name varchar(50)
last_name varchar(50)
email varchar(150)
last_updated timestamp default=localtimestamp

category
-
category_id varchar(10) PK
category varchar(20)
last_updated timestamp default=localtimestamp

subcategory
-
subcategory_id varchar(10) PK
subcategory varchar(20)
last_updated timestamp default=localtimestamp
